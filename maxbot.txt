import random
import telebot
TOKEN = 'ваш_токен_от_бота'
bot = telebot.TeleBot(TOKEN)
insults = [
    "Макс опять завис? Ничего нового...",
    "Эта программа настолько медленная, что даже черепаха бы её обогнала",
    "Макс — это как калькулятор на калькуляторе",
    "Когда уже научатся делать что-то нормальное, а не это недоразумение?",
    "Интерфейс Макса настолько неудобный, что даже слепой найдёт его проще",
    "Программа Макса — это как вечный двигатель, только вечный сбой",
    "Макс настолько глючный, что даже вирусы им брезгуют",
    "Когда уже научатся делать что-то, что не будет вылетать каждые 5 минут?",
    "Макс — это как чёрный ящик, только без содержимого",
    "Программа настолько устаревшая, что динозавры ещё пользовались ей"
]
@bot.message_handler(commands=['start'])
def start_message(message):
    bot.reply_to(message, "Привет! Я бот-критик программы Макс. Готов поливать её грязью 24/7!")

@bot.message_handler(func=lambda message: True)
def handle_message(message):
    # Выбираем случайную критическую фразу
    random_insult = random.choice(insults)
    bot.reply_to(message, random_insult)

if __name__ == '__main__':
    bot.infinity_polling()
